-for these commands, i have to be in folder
    cd backend/python

-to create the virtual environment for python
    python3 -m venv .venv

-to activate the venv
    -for macos/linux:
        source .venv/bin/activate

-to install packages from requirements.txt
    pip install -r requirements.txt

-to upgrade venv\python version
    .venv/bin/python3 -m pip install --upgrade pip

-to run the backend:
    python3 controller.py
